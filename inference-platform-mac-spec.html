Extensibility — adding open-source models later (Phase 09)

▾

The architecture is designed for this from day one. Adding vLLM on a G2 VM requires only two changes:

docker-compose.g2.yml (new overlay file)
services:
  vllm:
    image: vllm/vllm-openai:v0.6.3
    deploy:
      resources:
        reservations:
          devices: [{driver: nvidia, count: 1, capabilities: [gpu]}]
    command: ["--model", "meta-llama/Meta-Llama-3.1-8B-Instruct", "--port", "8000"]
Add to services/litellm/config.yaml (one new entry)
  - model_name: llama-3.1-8b
    litellm_params:
      model: openai/llama-3.1-8b      # openai/ prefix = OpenAI-compatible endpoint
      api_base: http://vllm:8000/v1
      api_key: none
      metadata: {provider: self-hosted, tier: standard}

Nothing else changes. Kong, guardrails, auth, observability, governance — all work identically with the new model.

Kong sits in front of LiteLLM as the edge gateway. After this phase clients only talk to Kong :8080. LiteLLM :4000 is removed from the host network — internal only. Kong enforces rate limiting, API key auth, request correlation, WAF, and API versioning.

from fastapi import FastAPI, HTTPException, Request
import httpx, json, hashlib, time
from datetime import datetime

app = FastAPI()

@app.post("/v1/chat/completions")
async def guarded_chat(req: Request):
    body    = await req.json()
    api_key = req.headers.get("Authorization","").replace("Bearer ","")
    req_id  = req.headers.get("X-Request-ID","unknown")
    text    = " ".join(m.get("content","") for m in body.get("messages",[])
                       if isinstance(m.get("content"), str))

    # ── 1. PII detection and redaction ─────────────────────────────
    async with httpx.AsyncClient(timeout=10) as c:
        r = await c.post("http://presidio-analyzer:3000/analyze",
                         json={"text": text, "language": "en"})
        entities = r.json()
    if entities:
        async with httpx.AsyncClient(timeout=10) as c:
            r = await c.post("http://presidio-anonymizer:3000/anonymize",
                             json={"text": text, "analyzer_results": entities,
                                   "anonymizers":{"DEFAULT":{"type":"replace","new_value":""}}})
            redacted = r.json()["text"]
        for msg in body.get("messages",[]):
            if isinstance(msg.get("content"), str):
                msg["content"] = redacted

    # ── 2. LLM Guard input scan ────────────────────────────────────
    async with httpx.AsyncClient(timeout=15) as c:
        r = await c.post("http://llm-guard:8000/analyze/output",
                         json={"prompt": text, "output": ""})
        guard = r.json()
    if not guard.get("is_valid", True):
        await audit("BLOCKED_PRE", req_id, api_key, body.get("model",""), len(entities))
        raise HTTPException(400, {"error":"request_blocked","detail":guard.get("scanners_results",{})})

    # ── 3. OPA policy check ────────────────────────────────────────
    async with httpx.AsyncClient(timeout=5) as c:
        r = await c.post("http://opa:8181/v1/data/inference/authz",
                         json={"input":{"path":"/v1/chat/completions","method":"POST",
                                        "body":{"model":body.get("model")},
                                        "user":{"roles":["model-user"],"team":"default"}}})
        if not r.json().get("result",{}).get("allow", False):
            raise HTTPException(403, {"error":"policy_violation"})

    await audit("REQUEST", req_id, api_key, body.get("model",""), len(entities))

    # ── 4. Forward to LiteLLM ──────────────────────────────────────
    async with httpx.AsyncClient(timeout=120) as c:
        resp = await c.post("http://litellm:4000/v1/chat/completions",
                            json=body, headers={"Authorization":f"Bearer {api_key}"})
    result = resp.json()

    # ── 5. LLM Guard output scan ───────────────────────────────────
    output = result.get("choices",[{}])[0].get("message",{}).get("content","")
    async with httpx.AsyncClient(timeout=15) as c:
        r = await c.post("http://llm-guard:8000/analyze/output",
                         json={"prompt": text, "output": output})
        guard_out = r.json()
    if not guard_out.get("is_valid", True):
        await audit("BLOCKED_POST", req_id, api_key, body.get("model",""), 0)
        raise HTTPException(451, {"error":"response_blocked"})

    await audit("RESPONSE", req_id, api_key, body.get("model",""), 0)
    return result

async def audit(event, req_id, api_key, model, pii_count):
    """Immutable audit entry — NEVER stores prompt content."""
    entry = {
        "timestamp":    datetime.utcnow().isoformat()+"Z",
        "event_type":   event,
        "request_id":   req_id,
        "key_hash":     hashlib.sha256(api_key.encode()).hexdigest()[:16],
        "model":        model,
        "pii_redacted": pii_count

#!/usr/bin/env python3

import pandas as pd, mlflow
from fairlearn.metrics import MetricFrame, demographic_parity_ratio
from sklearn.metrics import accuracy_score

mlflow.set_tracking_uri("http://localhost:5000")

def run_fairness_report(model_name: str, sample_file: str):
    """
    sample_file: JSONL with fields:
      {label, predicted_label, sensitive_attr_gender, sensitive_attr_age_group}
    Labels are task-specific (e.g. sentiment: positive/negative)
    """
    df = pd.read_json(sample_file, lines=True)
    for attr in ["sensitive_attr_gender", "sensitive_attr_age_group"]:
        if attr not in df.columns:
            continue
        mf = MetricFrame(
            metrics={"accuracy": accuracy_score},
            y_true=df["label"], y_pred=df["predicted_label"],
            sensitive_features=df[attr]
        )
        dp = demographic_parity_ratio(
            df["label"], df["predicted_label"], sensitive_features=df[attr]
        )
        print(f"Model: {model_name} | {attr} | DP ratio: {dp:.3f} (threshold: 0.80)")
        if dp < 0.80:
            print(f"  ⚠ ALERT: Below threshold for {attr}")
        with mlflow.start_run(run_name=f"fairness-{model_name}-{attr}"):
            mlflow.log_params({"model": model_name, "attribute": attr})
            mlflow.log_metric("demographic_parity_ratio", dp)
            mlflow.log_metric("overall_accuracy", mf.overall["accuracy"])
            mlflow.log_dict(mf.by_groups.to_dict(), "accuracy_by_group.json")

if __name__ == "__main__":
    run_fairness_report("gpt-4o-mini", "data/sample_responses.jsonl")
